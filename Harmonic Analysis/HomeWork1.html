    integer :: n, i

    real    :: deviation

subroutine Find_Abnormal(dat, n)
    implicit none
    integer :: n
    real    :: dat(n)
    real    :: diff(n-1)
    integer :: flag(n-1)
    real    :: max_diff = 0
    integer :: cur_flag = 0
    integer :: start_pos = 1
    integer :: cur_len
    integer :: min_len
    integer :: found_pos_1, found_pos_2
    integer :: i
!f2py intent(in, out, inplace) dat
!f2py integer,intent(hide),depend(dat) :: n=len(dat)
    cur_len = n
    min_len = n

    diff = dat(2:n) - dat(1:n-1)
    where(diff < 0.)
        flag = -1
    elsewhere
        flag = 1
    end where
    diff = abs(diff)
    do i = 1, n - 1
        if ( diff(i) > max_diff ) then
            max_diff = diff(i)
            found_pos_1 = i
        end if
    end do
    do i = 1, n - 1
        if (cur_flag /= flag(i)) then
            if (cur_len < min_len) then
                found_pos_2 = start_pos + 1
                min_len = cur_len
            end if
            start_pos = i
            cur_flag = flag(i)
            cur_len = 1
        else
            cur_len = cur_len + 1
        end if
    end do
    if ( found_pos_1 == found_pos_2 ) then
        write(*, "('Index of abnormal position in data:', I5)") found_pos_1
        dat(found_pos_1) = ( dat(found_pos_1 - 1) + dat(found_pos_1 + 1) ) / 2.;
    end if
    return
end subroutine

subroutine swap( a, b )
    implicit none
    real(kind=8) :: a, b, tmp
    tmp = a
    a = b
    b = tmp
    return
end subroutine swap

!高斯消元法解线性方程组
subroutine solve( A, b, x, n )
    implicit none
    integer         :: idx, i, j, k, n
    real(kind=8)    :: A(n, n), b(n), x(n)
    real(kind=8)    :: ssum, mmax, m
    real(kind=8)    :: eps = 1e-8
!f2py intent(in) A, b
!f2py intent(out) x
!f2py integer,intent(hide),depend(b) :: n=len(b)
    do k = 1, n - 1
        ! find the max
        mmax = dabs(A(k, k))
        idx = k
        do i = k + 1, n
            if ( dabs(A(i, k)) > mmax ) then
                mmax= dabs(A(i, k))
                idx = i
            end if
        end do
        ! handle the fail condition
        if ( mmax < eps ) then
            return
        end if
        ! swap
        if ( idx /= k ) then
            do j = 1, n
                call swap(A(k, j), A(idx, j))
            end do
            call swap(b(k), b(idx))
        end if
        ! elimination
        do i = k + 1, n
            if ( dabs(A(k, k)) < eps ) then
                return
            end if
            m = A(i, k) / A(k, k)
            do j = k, n
                A(i, j) = A(i, j) - A(k, j) * m
            end do
            b(i) = b(i) - b(k) * m
        end do
    end do
    ! back substitution
    do i = n, 1, -1
        ssum = 0.d0
        do j = i + 1, n
            ssum = ssum + A(i, j) * b(j)
        end do
        b(i) = ( b(i) - ssum ) / A(i, i)
    end do
    x = b
    return
end subroutine solve

调和分析的编程实现

思路分析

  1. 首先，为了计算调和常数，我们需要知道各个分潮的天文要素，如角速度、天文初相角、交点因子、交点订正角、以及差比关系等。
  2. 但是问题在于所有分潮的交点因子都是基于几个主要分潮的交点因子推算而得，尽管它们的关系是一定的，但是需要根据日期动态计算。
  3. 我们可以将Doodson数记录在数据文件中，让Fortran程序运行时直接读取并计算角速度；但是对于交点因子的推算关系，我们只能以公式的形式写出来。
  4. 考虑到这一部分的计算量非常小，因此直接用Python实现了一个简单的数据读取函数，根据日期和交点因子、交点订正角的推算关系，从分潮数据文件中计算并获取指定的各类参数。
  5. 这一模式主要是在模仿广泛使用的调和分析工具箱T_Tides，目的是使得程序具有更好的通用性，不用重新编译Fortran程序，只需要修改配置文件就可以直接使用任意的分潮进行调和分析。
  6. 而真正需要解高阶线性方程组并迭代求解随从分潮调和常数的计算密集部分，将使用Fortran编写。
  7. 用户需要保证自行给出对应于中间数据的正确时间，并保证数据个数为奇数。

def get_sigma_V(raw_data, sigma_const, V0_const, t):

            Doodson[4] * sigma_const[4] + \
            Doodson[5] * sigma_const[5]

         Doodson[4] * V0_const[4] + \
         Doodson[5] * V0_const[5] + \

    return sigma, V0 + sigma * t

    u = array(u)
    kappa = array(kappa)

    integer         :: i, j, n, mid
    !各种临时变量
    real(kind=8)    :: A_ij, B_ij, A_0i, A_0j, F_0, F_i, G_i, m_max
    !固定常数赋值
    real(kind=8)    :: pi = 3.14159265358979323846d0, eps = 1d-9

!下面是编译指导语句，用于给F2PY读取
!f2py intent(in,copy) P, zeta, sigma, V, f, kappa, alpha, corr
!f2py intent(out) H, g, S_0
!f2py integer,intent(hide),depend(zeta) :: M=len(zeta)
!f2py integer,intent(hide),depend(corr) :: Q=len(corr)

    !对η和ξ进行赋值
    do i = 1, P + Q
        eta(i) = f(i) * dsin(V(i))
         xi(i) = f(i) * dcos(V(i))
    end do

    !给随从分潮的a、b赋初始值，这里取零
    res_a(P + 1:P + Q) = 0.d0
    res_b(P + 1:P + Q) = 0.d0

    !记录前一次的解，用于确定主分潮的调和常数改变了多少；初始赋值为零
    bak_S_0 = 0.d0
    res_bak_a = 0.d0
    res_bak_b = 0.d0
    do while ( .true. )
        !对方程组左侧系数进行赋值
        !约定：前P个未知数为a_i，后P个未知数为b_i，最后一个为平均海面
        do i = 1, P
            A_0i = dsin(M*sigma(i)/2.d0) / dsin(sigma(i)/2.d0)
            A(i, 2*P + 1) = xi(i) * A_0i
            A(i + P, 2*P + 1) = eta(i) * A_0i
            do j = 1, P
                if ( i /= j ) then
                    A_ij = (dsin((sigma(i)-sigma(j))*M/2.d0) / dsin((sigma(i)-sigma(j))/2.d0) + &
                            dsin((sigma(i)+sigma(j))*M/2.d0) / dsin((sigma(i)+sigma(j))/2.d0)) / 2.d0
                    B_ij = (dsin((sigma(i)-sigma(j))*M/2.d0) / dsin((sigma(i)-sigma(j))/2.d0) - &
                            dsin((sigma(i)+sigma(j))*M/2.d0) / dsin((sigma(i)+sigma(j))/2.d0)) / 2.d0
                else
                    A_ij = ( M + dsin(M*sigma(i)) / dsin(sigma(i)) ) / 2.d0
                    B_ij = ( M - dsin(M*sigma(i)) / dsin(sigma(i)) ) / 2.d0
                end if
                A(i, j)         = xi(i) * xi(j) * A_ij + eta(i) * eta(j) * B_ij
                A(i, j + P)     = xi(i) * eta(j) * A_ij - eta(i) * xi(j) * B_ij
                A(i + P, j)     = eta(i) * xi(j) * A_ij - xi(i) * eta(j) * B_ij
                A(i + P, j + P) = eta(i) * eta(j) * A_ij + xi(i) * xi(j) * B_ij
            end do
        end do

        do j = 1, P
            A_0j = dsin(M*sigma(j)/2.d0) / dsin(sigma(j)/2.d0)
            A(2*P + 1, j) = xi(j) * A_0j
            A(2*P + 1, j + P) = eta(j) * A_0j
        end do
        A(2*P + 1, 2*P + 1) = M

        !计算数据的中间值，后面用于修正偏移
        mid = ( M - 1 ) / 2 + 1

        !对方程组右侧进行赋值
        F_0 = 0.d0
        do i = 1, M
            F_0 = F_0 + zeta(i)
        end do
        do i = 1, P
            F_i = 0.d0
            G_i = 0.d0
            do n = 1, M
                F_i = F_i + zeta(n) * dcos((n - mid)*sigma(i))
                G_i = G_i + zeta(n) * dsin((n - mid)*sigma(i))
            end do
            b(i) = xi(i) * F_i - eta(i) * G_i
            b(i + P) = eta(i) * F_i + xi(i) * G_i
            do j = P + 1, P + Q
                A_ij = (dsin((sigma(i)-sigma(j))*M/2.d0) / dsin((sigma(i)-sigma(j))/2.d0) + &
                        dsin((sigma(i)+sigma(j))*M/2.d0) / dsin((sigma(i)+sigma(j))/2.d0)) / 2.d0
                B_ij = (dsin((sigma(i)-sigma(j))*M/2.d0) / dsin((sigma(i)-sigma(j))/2.d0) - &
                        dsin((sigma(i)+sigma(j))*M/2.d0) / dsin((sigma(i)+sigma(j))/2.d0)) / 2.d0
                b(i)     = b(i)     - res_a(j) * ( xi(i) * xi(j) * A_ij + eta(i) * eta(j) * B_ij ) &
                                    - res_b(j) * ( xi(i) * eta(j) * A_ij - eta(i) * xi(j) * B_ij )
                b(i + P) = b(i + P) - res_a(j) * ( eta(i) * xi(j) * A_ij - xi(i) * eta(j) * B_ij ) &
                                    - res_b(j) * ( eta(i) * eta(j) * A_ij + xi(i) * xi(j) * B_ij )
            end do
        end do
        b(2*P + 1) = F_0
        do j = P + 1, P + Q
            A_0j = dsin(M*sigma(j)/2.d0) / dsin(sigma(j)/2.d0)
            b(2*P + 1) = b(2*P + 1) - res_a(j) * xi(j) * A_0j - res_b(j) * eta(j) * A_0j
        end do

        !调用高斯消元法模块求解线性方程组，方程中前P项赋值给a，后P项赋值给b，最后一项赋值给S_0
        call solve(A, b, x, 2*P + 1)
        res_a(1:P) = x(1:P)
        res_b(1:P) = x(P + 1:2 * P)
        S_0 = x(2*P + 1)

        m_max = dabs(S_0 - bak_S_0)
        do i = 1, P
            m_max = max(m_max, dabs(res_a(i) - res_bak_a(i)))
            m_max = max(m_max, dabs(res_b(i) - res_bak_b(i)))
        end do
        !如果迭代基本收敛，则退出循环，否则保存当前的解
        write(*,'("Current relative error: ", G17.9)') m_max
        if ( m_max < eps ) exit

        bak_S_0 = S_0
        res_bak_a = res_a(1:P)
        res_bak_b = res_b(1:P)

        !利用差比关系更新随从分潮的a、b
        do i = 1, Q
            res_a(i + P) = kappa(i) * ( res_a(corr(i)) * dcos(alpha(i)) - &
                                        res_b(corr(i)) * dsin(alpha(i)) )
            res_b(i + P) = kappa(i) * ( res_a(corr(i)) * dsin(alpha(i)) + &
                                        res_b(corr(i)) * dcos(alpha(i)) )
        end do
    end do

    !计算调和常数
    do i = 1, P + Q
        H(i) = dsqrt(res_a(i)**2 + res_b(i)**2)
        g(i) = datan(res_b(i) / res_a(i))
        !这一步是修正区时专用迟角g的取值范围，因为b和sin(g)显然应该同号
        if ( res_b(i) * dsin(g(i)) < 0.d0 ) g(i) = g(i) + pi
        if ( g(i) < 0.d0 ) g(i) = g(i) + 2 * pi
    end do

    return
end subroutine

subroutine predict( head, tail, num, zeta, sigma, V, f, H, g, S_0, mask, mask_num, delta )
    implicit none
    integer         :: head, tail, num, mask_num, i, j, idx
    real(kind=8)    :: zeta(tail - head + 1), sigma(num), V(num), f(num), H(num), g(num), S_0
    integer         :: mask(mask_num)
    real(kind=8)    :: t, delta

!f2py intent(in,copy) head, tail, sigma, V, f, H, g, mask, delta, S_0
!f2py intent(out) zeta
!f2py integer,intent(hide),depend(sigma) :: num=len(sigma)
!f2py integer,intent(hide),depend(mask) :: mask_num=len(mask)

    t = head * delta
    do i = 1, tail - head + 1
        zeta(i) = S_0
        do j = 1, mask_num
            idx = mask(j)
            zeta(i) = zeta(i) + f(idx) * H(idx) * dcos(sigma(idx) * t + V(idx) - g(idx))
        end do
        t = t + delta
    end do
    return
end subroutine

潮汐资料调和分析工具使用说明

  1. 按照指定格式编辑配置文件constituent.ini，输入选定的分潮；已将课本P87页表格中所有分潮写入该文件。
  2. 用数据中间时刻的对应时间，以及需要用于预报的分潮名称来调用数据读取工具getData，获取这些参数：主分潮数量(P)、随从分潮数量(Q)、角速度(sigma)、初相角(V)、交点因子(f)、振幅比(kappa)、迟角差(alpha)、主分潮和随从分潮的对应关系(corr)、选取用来做预报的分潮编号(mask)、分潮数字编号与名称的对应关系(record)。
    1. 在上述过程中，用于预报的分潮名称可以省略，此时默认使用四大分潮和两个浅水分潮进行预报，如下命令所示： > P, Q, sigma, V, f, kappa, alpha, corr, mask, record = get_variables(middle_date)
    2. 如果输入一个空列表，则会自动使用所有分潮进行潮位预报，如下命令所示： > P, Q, sigma, V, f, kappa, alpha, corr, mask, record = get_variables(middle_date, [])
    3. 如果输入其他有效参数，则使用指定的分潮进行潮位预报，如下命令所示： > P, Q, sigma, V, f, kappa, alpha, corr, mask, record = get_variables(middle_date, ['m_2', 's_2', 'k_1', 'o_1', 'm_4', 'ms_4'])
    4. 最后还有一个隐藏参数，用于指定所计算海区的\(\Delta g\)取值，如下命令所示： > P, Q, sigma, V, f, kappa, alpha, corr, mask, record = get_variables(middle_date, ['m_2', 's_2', 'k_1', 'o_1', 'm_4', 'ms_4'], delta_g = 50.)
  3. 以这些参数调用调和分析计算模块，用以计算振幅\(H\)、迟角\(g\)和平均海面\(S_{0}\)，如下命令所示： > H, g, S_0 = Tides.analyze(P, zeta, sigma, V, f, kappa, alpha, corr)
  4. 继续使用上述参数和调和常数调用潮位预测模块，用以回报潮位过程曲线。此时需要指定开始与结束的时间点（均为相对于原始数据中间时刻的偏移）和数据的时间间隔。
    1. 下述命令回报的范围为数据范围，即从相对于中间时刻第-371个数据到相对于中间时刻的第371个数据，时间间隔为1.0小时： > zeta_predict = Tides.predict(-371, 371, sigma, V, f, H, g, S_0, mask, 1.)
    2. 下面的命令由于时间间隔改为了1/60小时，所以数据个数就需要对应乘以60： > zeta_predict_minute = Tides.predict(-371 * 60, 371 * 60, sigma, V, f, H, g, S_0, mask, 1./60.)

工具介绍

Python

Python作为一种高度成熟且开源免费的科学计算语言，在海洋学领域的应用尚不广泛，这里对其进行简单的介绍。
主要特点：

  • 语法简明清晰：代码直观且可读性强，项目规模的增大不会显著降低其可维护性。
  • 表现能力出色：实现功能所需的代码量很短，这可以有效减少BUG，易于编写。

技术细节：

  • 动态类型：如同MATLAB一样定义变量时不需要声明其类型，同样也不需要声明函数参数、返回值的类型。
  • 自动垃圾回收：如同Java一样不需要为对象和数组显式地分配或回收内存，不会导致任何内存泄漏的问题。
  • 解释型语言：不需要对代码进行编译，直接解释执行即可（执行时会自动编译成字节码，类似MATLAB的JIT技术）。

优势：

  • 将Python引入海洋学研究中的最大优势在于简化了编程的复杂度，避免将精力消耗在语言细节上。
  • 具有卓越的语法设计，相比于MATLAB等简陋的过程化编程，其现代化的面向对象思想以及设计完善的包管理机制最大限度地提升了代码的可重用性和可维护性。
  • 本身还具有胶水语言的特性，可以很方便地与C、Fortran等高性能计算语言进行混合编程，不仅弥补了自身运行效率的不足，同时也简化了编程工作——可以直接复用。
  • 拥有数目巨大的各类库函数，以及开源社区的大力支持，可以用它实现几乎一切编程需求——数据读取、处理、绘图、编写用户图形界面等所有工作都可以用Python完成。

缺陷：

  • 相比于C、Fortran等编译型语言，Python等动态解释型语言的最大缺陷就是运行效率较差。不过上面已经提到过，混合编程可以弥补这一缺陷。
  • 各种函数库本身有些分散化，在不同的环境下文档和所实现的功能会略有差异，可能会导致一些兼容性问题。

Python的科学计算方面的内容由许多库构成，下面介绍几个常用的函数库。

Numpy

  • NumPy为Python带来了真正的多维数组功能，并且提供了丰富的函数库处理这些数组。它提供了Array对象将常用的数学函数都进行数组化，使得这些数学函数能够直接对数组进行操作，将本来需要在Python级别进行的循环，放到C语言的运算中，从而显著提高了程序的运算速度。
  • 由于原本基于Array进行运算的现有函数就极多，因此为了更好地与其他函数进行功能整合，作业中也会引入Array对象来存储数据。

Scipy

Scipy提供基于Array对象的高级计算函数，核心就是是一些久经考验的Fortran数值计算库，例如：

  • 线性代数使用LAPACK库
  • 快速傅立叶变换使用FFTPACK库
  • 常微分方程求解使用ODEPACK库
  • 非线性方程组求解以及最小值求解等使用MINPACK库

MatPlotLib

  • MatPlotLib是Python最著名的绘图库，提供了一整套和MATLAB相似的命令API，十分适合交互式地进行制图。
  • 它提供了两种绘图风格，MATLAB风格以及面向对象风格。前者适合于简单进行绘图，而后者适合用于绘制布局复杂的图像，可以针对图像中的每一个元素进行十分精确的调节。

F2PY

  • F2PY是一个可以自动解析Fortran代码并将其编译为Python模块的程序，编译好的模块可以用import语句导入Python运行环境。
  • 这个程序是NumPy的一部分，但是在使用时环境中必须包含一个支持的Fortran编译器。
  • 所有作业使用Windows平台下Mingw项目提供的gfortran对Fortran代码进行编译。
